183  int len = UR_DEFAULT_LENGTH_OF_TEMPLATE, act_len = 0;

295 const char *ur_get_type_and_name_from_string(const char *source, char **name, char **type, int *length_name, int *length_type)
296 {
297  int length_type_2 = 0, length_name_2 = 0;
298  const char *source_cpy;
299  /* skip white spaces */
300  while (*source != 0 && isspace(*source)) {
301  source++;
302  }
303  /* start of type */
304  source_cpy = source;
305  while (*source != 0 && !isspace(*source)) {
306  length_type_2++;
307  source++;
308  }
309  /* end of type */
310 
311  /* copy "type" string (realloc destination if needed) */
312  if (length_type_2 >= *length_type) {
313  if (*type != NULL) {
314  free(*type);
315  }
316  *type = (char *) malloc(sizeof(char) * (length_type_2 + 1));
317  if (*type == NULL) {
318  return NULL;
319  }
320  *length_type = length_type_2 + 1;
321  }
322  memcpy(*type, source_cpy, length_type_2);
323  (*type)[length_type_2] = 0;
324 
325  /* skip white spaces */
326  while (*source != 0 && isspace(*source)) {
327  source++;
328  }
329  /* start of name */
330  source_cpy = source;
331  while (*source != 0 && !isspace(*source) && *source != ',') {
332  length_name_2++;
333  source++;
334  }
335  /* end of name */
336 
337  /* copy "name" string (realloc destination if needed) */
338  if (length_name_2 >= *length_name) {
339  if (*name != NULL) {
340  free(*name);
341  }
342  *name = (char *) malloc(sizeof(char) * (length_name_2 + 1));
343  if (*name == NULL) {
344  return NULL;
345  }
346  *length_name = length_name_2 + 1;
347  }
348  memcpy(*name, source_cpy, length_name_2);
349  (*name)[length_name_2] = 0;
350  /* skip white spaces */
351  while (*source != 0 && isspace(*source)) {
352  source++;
353  }
354  /* skip comma */
355  if (*source == ',') {
356  source++;
357  }
358  return source;

563 int ur_define_field(const char *name, ur_field_type_t type)
564 {
565  int insert_id;
566  char * name_copy;
567  int name_len;
568  if (name == NULL) {
569  return UR_E_INVALID_NAME;
570  }
571  //check the regural expression of a name
572  name_len = strlen(name);
573  if (name_len == 0) {
574  return UR_E_INVALID_NAME;
575  }
576  if (!((name[0] >= 'A' && name[0] <= 'Z') || (name[0] >= 'a' && name[0] <= 'z'))) {
577  return UR_E_INVALID_NAME;
578  }
579  for (int i = 1; i < name_len; i++) {
580  if (!((name[i] >= 'A' && name[i] <= 'Z') || (name[i] >= 'a' && name[i] <= 'z') || (name[i] >= '0' && name[i] <= '9') || name[i] == '_')) {
581  return UR_E_INVALID_NAME;
582  }
583  }
584  if (ur_field_specs.ur_allocated_fields == ur_field_specs.ur_last_statically_defined_id) {
585  int init_val = ur_init(UR_FIELD_SPECS_STATIC);
586  if (init_val != 0) {
587  return init_val;
588  }
589  }
590  //check if the field is already defined
591  for (int i = 0; i < ur_field_specs.ur_last_id; i++) {
592  if (ur_field_specs.ur_field_names[i] != NULL && strcmp(name, ur_field_specs.ur_field_names[i]) == 0) {
593  if (type == ur_field_specs.ur_field_types[i]) {
594  //name exists and type is equal
595  return i;
596  } else {
597  //name exists, but type is different
598  return UR_E_TYPE_MISMATCH;
599  }
600  }
601  }
602  //create new field
603  name_copy = (char *) calloc(sizeof(char), strlen(name) + 1);

826 ur_template_t *ur_create_template(const char *fields, char **errstr)
827 {
828  // Count number of fields
829  int n_fields = 0, written_fields = 0;
830  if (fields) {
831  /* skip leading spaces */
832  while (*fields != '\0' && isspace(*fields)) {
833  fields++;
834  }
835  /* Count number of fields */
836  if (*fields != '\0') {
837  n_fields = 1;
838  const char *tmp = fields;
839  while (*tmp != '\0') {
840  if (*(tmp++) == ',') {
841  n_fields++;
842  }
843  }
844  }
845  }
846  // Allocate array of field_spec structs
847  field_spec_t *fields_spec = malloc(n_fields * sizeof(field_spec_t));
848  if (fields_spec == NULL && n_fields > 0) {
849  if (errstr != NULL) {
850  *errstr = (char *) malloc(strlen(UR_MEMORY_ERROR) + 1);
851  if (*errstr != NULL) {
852  strcpy(*errstr, UR_MEMORY_ERROR);
853  }
854  }
855  return NULL;
856  }
857  // Parse fields and fill the array
858  const char *start_ptr = fields;
859  const char *end_ptr;
860  for (int i = 0; i < n_fields; i++) {
861  // Get field name
862  end_ptr = start_ptr;
863  /* go to the first space / comma / end-of-string */
864  while (!isspace(*end_ptr) && *end_ptr != ',' && *end_ptr != '\0') {
865  end_ptr++;
866  }
867  int len = end_ptr - start_ptr;
868  fields_spec[written_fields].name = malloc(len + 1);
869  if (fields_spec[written_fields].name == NULL) {
870  if (errstr != NULL) {
871  *errstr = (char *) malloc(strlen(UR_MEMORY_ERROR) + 1);
872  if (*errstr != NULL) {
873  strcpy(*errstr, UR_MEMORY_ERROR);
874  }
875  }
876  for (int j = 0; j < i; j++) {
877  free(fields_spec[j].name);
878  }
879  free(fields_spec);
880  return NULL;
881  }
882  memcpy(fields_spec[written_fields].name, start_ptr, len);
883  fields_spec[written_fields].name[len] = 0;
884  start_ptr = end_ptr;
885  while ((isspace(*start_ptr) || *start_ptr == ',') && *start_ptr != '\0') {
886  start_ptr++;
887  }
888  // Get field ID
889  int id_by_name = ur_get_id_by_name(fields_spec[written_fields].name);
890  if (id_by_name == UR_E_INVALID_NAME) {
891  // Unknown field name
892  if (errstr != NULL) {
893  *errstr = (char *) malloc(100);
894  if (*errstr != NULL) {
895  int n;
896  n = snprintf(*errstr, 100, "field: %s is not defined.", fields_spec[written_fields].name);
897  if (n >= 100) {
898  strcpy(*errstr, "given field is not defined");
899  }
900  }
901  }
902  for (int j = 0; j <= written_fields; j++) {
903  free(fields_spec[j].name);
904  }
905  free(fields_spec);
906  return NULL;
907  }
908  //check if the field is not in the template.
909  int in_the_template = 0;
910  for (int j = 0; j < written_fields; j++) {
911  if (fields_spec[j].id == id_by_name) {
912  in_the_template = 1;
913  break;
914  }
915  }
916  //if the field is not already int the template, copy values and move the index, otherwise just free the string with name.
917  if (in_the_template == 0) {
918  fields_spec[written_fields].id = id_by_name;
919  // Get field size
920  fields_spec[written_fields].size = ur_get_size(fields_spec[written_fields].id);
921  written_fields++;
922  } else {
923  free(fields_spec[written_fields].name);
924  fields_spec[written_fields].name = NULL;
925  }
926  }
927  // Sort fields according to UniRec specification (by size and names)
928  if (n_fields > 0) {
929  qsort(fields_spec, written_fields, sizeof(field_spec_t), compare_fields);
930  }
931  // Allocate memory for the template
932  ur_template_t *tmplt = (ur_template_t *) calloc(sizeof(ur_template_t), 1);
933  if (tmplt == NULL) {
934  for (int i = 0; i < written_fields; i++) {
935  free(fields_spec[i].name);
936  }
937  free(fields_spec);
938  if (errstr != NULL) {
939  *errstr = (char *) malloc(strlen(UR_MEMORY_ERROR) + 1);
940  if (*errstr != NULL) {
941  strcpy(*errstr, UR_MEMORY_ERROR);
942  }
943  }
944  return NULL;
945  }
946  //set no direction to the template
947  tmplt->direction = UR_TMPLT_DIRECTION_NO;
948  //allocate memory for offset table
949  tmplt->offset_size = ur_field_specs.ur_last_id;
950  tmplt->offset = malloc(ur_field_specs.ur_last_id * sizeof(uint16_t));
951  if (tmplt->offset == NULL) {
952  for (int i = 0; i < written_fields; i++) {
953  free(fields_spec[i].name);
954  }
955  free(fields_spec);
956  free(tmplt);
957  if (errstr != NULL) {
958  *errstr = (char *) malloc(strlen(UR_MEMORY_ERROR) + 1);
959  if (*errstr != NULL) {
960  strcpy(*errstr, UR_MEMORY_ERROR);
961  }
962  }
963  return NULL;
964  }
965  // Set all fields to invalid offset
966  memset(tmplt->offset, 0xff, ur_field_specs.ur_last_id * sizeof(uint16_t));
967  // Fill offsets of all fields into the table
968  uint16_t offset = 0;
969  uint16_t first_dynamic = UR_NO_DYNAMIC_VALUES;
970  for (int i = 0; i < written_fields; i++) {
971  // Check whether offset is not already set
972  if (tmplt->offset[fields_spec[i].id] != 0xffff) {
973  // Offset is already set - duplicated field
974  for (int i = 0; i < written_fields; i++) {
975  free(fields_spec[i].name);
976  }
977  free(fields_spec);
978  free(tmplt);
979  if (errstr != NULL) {
980  *errstr = (char *) malloc(strlen(UR_MEMORY_ERROR) + 1);
981  if (*errstr != NULL) {
982  strcpy(*errstr, UR_MEMORY_ERROR);
983  }
984  }
985  return NULL;
986  }
987  // Set offset
988  if (fields_spec[i].size < 0) { // dynamic field
989  tmplt->offset[fields_spec[i].id] = offset;
990  offset += 4;
991  if (first_dynamic == UR_NO_DYNAMIC_VALUES) {
992  first_dynamic = i;
993  }
994  } else { // static field
995  tmplt->offset[fields_spec[i].id] = offset;
996  offset += fields_spec[i].size;
997  }
998  }
999  tmplt->first_dynamic = first_dynamic;
1000  tmplt->static_size = offset;
1001 
1002  //save ids to template
1003  tmplt->ids = (ur_field_id_t *) malloc(sizeof(ur_field_id_t) * written_fields);
1004  if (tmplt->ids == NULL) {
1005  for (int i = 0; i < written_fields; i++) {
1006  free(fields_spec[i].name);
1007  }
1008  free(fields_spec);
1009  free(tmplt);
1010  if (errstr != NULL) {
1011  *errstr = (char *) malloc(strlen(UR_MEMORY_ERROR) + 1);
1012  if (*errstr != NULL) {
1013  strcpy(*errstr, UR_MEMORY_ERROR);
1014  }
1015  }
1016  return NULL;
1017  }
1018  tmplt->count = written_fields;
1019  for (int i = 0; i < written_fields; i++) {
1020  tmplt->ids[i] = fields_spec[i].id;
1021  }
1022  // Free array of field specs
1023  for (int i = 0; i < written_fields; i++) {
1024  free(fields_spec[i].name);
1025  }
1026  free(fields_spec);
1027  return tmplt;
1028 }
