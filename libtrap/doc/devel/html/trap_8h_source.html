141 /**
142  * \defgroup trapifcspec Specifier of TRAP interfaces
143  * The format of -i parameter of modules sets up TRAP interfaces.
144  * #TRAP_IFC_DELIMITER divides specifier into interfaces.
145  * #TRAP_IFC_PARAM_DELIMITER divides parameters of one interface.
146  *
147  * Each TRAP interface must have the type (\ref ifctypes) as the first parameter.
148  * The following parameters are passed to the interface and they are interface-dependent.
149  *
150  * The format example, let's assume the module has 1 input IFC and 1 output IFC:
151  *
152  * <BLOCKQUOTE>-i t:localhost:7600,u:my_socket</BLOCKQUOTE>
153  *
154  * This sets the input IFC to TCP type and it will connect to localhost, port 7600.
155  * The output IFC will listen on UNIX socket with identifier my_socket.
156  *
157  * @{

872  */
873 #ifdef HAVE_SIGACTION
874 #define TRAP_REGISTER_DEFAULT_SIGNAL_HANDLER() \
875  do {\
876  if (trap_get_verbose_level() >= 1)\
877  printf("Setting signal handler for SIGINT and SIGTERM using 'sigaction' function.\n");\
878  struct sigaction act;\
879  /* Set default signal handler function*/\
880  act.sa_handler = trap_default_signal_handler;\
881  act.sa_flags = 0;\

903  * Generates code that parses command-line parameters using trap_parse_params,
904  * intializes TRAP library using trap_init and handle possible errors.
905  * It calls exit(1) when an error has occured.
906  * Place this macro at the beginning of your main function.
907  * \param[in,out] argc Number of command-line parameters.
