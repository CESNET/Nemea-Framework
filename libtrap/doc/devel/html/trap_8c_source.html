204 /**
205  * \brief Check content of buffer, iterate over message headers
206  * \param [in] buffer start of buffer
207  * \param [in] buffer_size size of buffer
208  * \return 0 on success, number of errors otherwise
209  */
210 int trap_check_buffer_content(void *buffer, uint32_t buffer_size)
211 {
212  uint32_t offset, check_mess_counter = 0;
213  uint16_t *check_mess_header;
214  int errors = 0;
215  void *check_mess_pointer;
216  for (offset = 0, check_mess_header = check_mess_pointer = buffer;
217  ((offset < buffer_size) && (offset < TRAP_IFC_MESSAGEQ_SIZE));) {
218  check_mess_counter++;
219  /* go to next size, skip header + payload */
220  offset += sizeof(*check_mess_header) + (*check_mess_header);
221  check_mess_pointer += sizeof(*check_mess_header) + (*check_mess_header);
222  check_mess_header = (uint16_t *) check_mess_pointer;
223  }
224  if (offset != buffer_size) {
225  VERBOSE(CL_ERROR, "Not enough data or some headers are malformed.");
226  errors++;
227  return errors;
228  }
229  return errors;
230 }

241 static inline int trap_read_from_buffer(trap_ctx_priv_t *ctx, uint32_t ifc_idx, const void **data, uint16_t *size, int timeout)
242 {
243  int result = TRAP_E_TIMEOUT;
244  /* pointer to current message header */
245  uint32_t tempbufheader = 0;
246 
247  /* pointer to current message payload */
248  void *bp = ctx->in_ifc_list[ifc_idx].buffer;
249  pthread_mutex_lock(&ctx->in_ifc_list[ifc_idx].ifc_mtx);
250  if ((ctx->in_ifc_list[ifc_idx].buffer_full == 0) || (ctx->in_ifc_list[ifc_idx].buffer_full > TRAP_IFC_MESSAGEQ_SIZE)) {
251  /* get new data and store into buffer, set buffer_full size */
252  ctx->in_ifc_list[ifc_idx].buffer_pointer = ctx->in_ifc_list[ifc_idx].buffer;
253  result = ctx->in_ifc_list[ifc_idx].recv(ctx->in_ifc_list[ifc_idx].priv, bp, &tempbufheader, timeout);
254  if (result == TRAP_E_FORMAT_MISMATCH) {
255  goto exit;
256  }
257 #ifdef BUFFERING_CHECK_HEADERS
258  if (trap_check_buffer_content(bp, tempbufheader) != 0) {
259  VERBOSE(CL_ERROR, "Buffer is not valid.");
260  }
261 #endif
262  if (result == TRAP_E_OK) {
263  ctx->counter_recv_buffer[ifc_idx]++;
264 
265  ctx->in_ifc_list[ifc_idx].buffer_full = tempbufheader;
266  ctx->in_ifc_list[ifc_idx].buffer_pointer = ctx->in_ifc_list[ifc_idx].buffer;
267  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "read received new buffer new bf %"PRIu32" %p",
268  ctx->in_ifc_list[ifc_idx].buffer_full,
269  ctx->in_ifc_list[ifc_idx].buffer_pointer));
270  #ifdef TESTBUFFERING
271  VERBOSE(CL_VERBOSE_OFF, "Received buffer of size %u.", ctx->in_ifc_list[ifc_idx].buffer_full);
272  #endif
273  } else {
274  goto exit;
275  }
276  }
277 
278  if (ctx->in_ifc_list[ifc_idx].buffer_full > 0) {
279  /* get message from buffer */
280  (*size) = ntohs(*((uint16_t *) ctx->in_ifc_list[ifc_idx].buffer_pointer));
281  (*data) = (ctx->in_ifc_list[ifc_idx].buffer_pointer + sizeof(*size));
282  /* decrease buffer_full size by returned payload and its header */
283  ctx->in_ifc_list[ifc_idx].buffer_full -= (*size + sizeof(*size));
284  ctx->in_ifc_list[ifc_idx].buffer_pointer += (*size) + sizeof(*size);
285  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "read from buffer %"PRIu16" B skip %"PRIu64" B, new bf %"PRIu32" %p",
286  (*size),
287  (*size + sizeof(*size)),
288  ctx->in_ifc_list[ifc_idx].buffer_full,
289  ctx->in_ifc_list[ifc_idx].buffer_pointer));
290  result = TRAP_E_OK;
291  } else {
292  (*size) = 0;
293  }
294 exit:
295  pthread_mutex_unlock(&ctx->in_ifc_list[ifc_idx].ifc_mtx);
296  if (result == TRAP_E_OK) {
297  ctx->counter_recv_message[ifc_idx]++;
298  if (ctx->in_ifc_list[ifc_idx].client_state == FMT_CHANGED) {
299  ctx->in_ifc_list[ifc_idx].client_state = FMT_OK;
300  return TRAP_E_FORMAT_CHANGED;
301  }
302  }
303  return result;
304 }

307 {
308  assert(priv->buffer_index <= (TRAP_IFC_MESSAGEQ_SIZE - sizeof(trap_buffer_header_t)));

316 static inline int trap_store_into_buffer(trap_ctx_priv_t *ctx, unsigned int ifc, const void *data, uint16_t size, int timeout, char flush)
317 {
318  /* Declaration of variables, we can have small buffer, initialization after checking the condition. */
319  uint32_t freespace, needed_size = size + sizeof(size);
320  int result;
321 
322  if (ctx->out_ifc_list[ifc].ifc_type == TRAP_IFC_TYPE_BLACKHOLE) {
323  return TRAP_E_OK;
324  }
325 
326  /* Can we put message at least into empty buffer? In the worst case, we could end up with SEGFAULT -> rather skip with error */
327  if (needed_size > TRAP_IFC_MESSAGEQ_SIZE) {
328  return trap_errorf(ctx, TRAP_E_MEMORY, "Buffer is too small for this message. Skipping...");
329  }
330 
331  if (flush != 0) {
332  /* Autoflush call, trying to lock section, maybe interface is waiting for clients -> rather skip than block the whole thread. */
333  if (pthread_mutex_trylock(&ctx->out_ifc_list[ifc].ifc_mtx) != 0) {
334  return TRAP_E_OK;
335  }
336  } else {
337  /* Lock this section at first before sending whole buffer. */
338  pthread_mutex_lock(&ctx->out_ifc_list[ifc].ifc_mtx);
339  }
340  /* initialization in locked section, otherwise autoflush can send buffer which has been already sent */
341  if (ctx->out_ifc_list[ifc].buffer_index <= (TRAP_IFC_MESSAGEQ_SIZE - sizeof(trap_buffer_header_t))) {
342  freespace = TRAP_IFC_MESSAGEQ_SIZE - ctx->out_ifc_list[ifc].buffer_index - sizeof(trap_buffer_header_t);
343  } else {
344  freespace = 0;
345  }
346  result = TRAP_E_TIMEOUT;
347 
348  /* Is this a autoflush call? If we have empty buffer, we do not send anything. */
349  if (flush != 0) {
350  if (ctx->out_ifc_list[ifc].buffer_index != 0) {
351 #ifdef BUFFERING_CHECK_HEADERS
352  if (trap_check_buffer_content(ctx->out_ifc_list[ifc].buffer, ctx->out_ifc_list[ifc].buffer_index) != 0) {
353  VERBOSE(CL_ERROR, "Buffer is not valid.");
354  }
355 #endif
356  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "sending by autoflush %"PRIu32" B from %p", ctx->out_ifc_list[ifc].buffer_index, ctx->out_ifc_list[ifc].buffer));
357 
358  ctx->out_ifc_list[ifc].buffer_occupied = 1;
359  trap_buffer_header_t *h = (trap_buffer_header_t *) ctx->out_ifc_list[ifc].buffer_header;
360  h->data_length = htonl(ctx->out_ifc_list[ifc].buffer_index);
361  result = ctx->out_ifc_list[ifc].send(ctx->out_ifc_list[ifc].priv, ctx->out_ifc_list[ifc].buffer_header,
362  ctx->out_ifc_list[ifc].buffer_index + sizeof(trap_buffer_header_t), timeout);
363 
364  if (result == TRAP_E_OK) {
365  ctx->counter_send_buffer[ifc]++;
366  ctx->out_ifc_list[ifc].buffer_index = 0;
367  ctx->out_ifc_list[ifc].buffer_occupied = 0;
368  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "Sending partial buffer invoked by autoflush timeout on interface %d", ifc));
369  } else {
370  VERBOSE(CL_VERBOSE_LIBRARY, "Autoflush was not successful.");
371  if (trap_ctx_get_client_count(ctx, ifc) == 0) {
372  ctx->out_ifc_list[ifc].buffer_occupied = 0;
373  }
374  }
375  }
376  goto fn_exit;
377  }
378  /* we send buffer before timeout, no need to flush it */
379  ctx->out_ifc_list[ifc].bufferflush = 1;
380 
381  if ((freespace >= needed_size) && (ctx->out_ifc_list[ifc].bufferswitch == 1)) {
382  /* we have enough space, buffering is enabled and size is not "flush" */
383 
384  insert_into_buffer(&ctx->out_ifc_list[ifc], data, size);
385 
386  result = TRAP_E_OK;
387 
388  } else {
389  /* not enough space */
390 
391 #ifdef BUFFERING_CHECK_HEADERS
392  if (trap_check_buffer_content(ctx->out_ifc_list[ifc].buffer, ctx->out_ifc_list[ifc].buffer_index) != 0) {
393  VERBOSE(CL_ERROR, "Buffer is not valid.");
394  }
395 #endif
396 
397  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "sending %"PRIu32" B from %p", ctx->out_ifc_list[ifc].buffer_index, ctx->out_ifc_list[ifc].buffer));
398 
399 #ifdef BUFFERING_CREATE_DUMPS
400  char *n = NULL;
401  if (asprintf(&n, "store-buffers-dump%04"PRIu64, ctx->counter_send_buffer[ifc]) != -1) {
402  mkdir(n, 0700);
403  ctx->out_ifc_list[ifc].create_dump(ctx->out_ifc_list[ifc].priv, ifc, n);
404  free(n);
405  }
406 #endif
407 
408  if (ctx->out_ifc_list[ifc].bufferswitch == 0) {
409  insert_into_buffer(&ctx->out_ifc_list[ifc], data, size);
410  }
411 
412  ctx->out_ifc_list[ifc].buffer_occupied = 1;
413  trap_buffer_header_t *h = (trap_buffer_header_t *) ctx->out_ifc_list[ifc].buffer_header;
414  h->data_length = htonl(ctx->out_ifc_list[ifc].buffer_index);
415  result = ctx->out_ifc_list[ifc].send(ctx->out_ifc_list[ifc].priv, ctx->out_ifc_list[ifc].buffer_header,
416  ctx->out_ifc_list[ifc].buffer_index + sizeof(trap_buffer_header_t), timeout);
417 
418  /* if the buffer was successfully sent OR we have no client: */
419  if (result == TRAP_E_OK || result == TRAP_E_IO_ERROR) {
420  if (result == TRAP_E_OK) {
421  /*
422  * buffer was successfully sent but we still have current message pending/not stored
423  * it will be the first message in buffer
424  */
425  ctx->counter_send_buffer[ifc]++;
426  } else {
427  /* we had no client but we can propagate either OK or TIMEOUT: */
428  result = TRAP_E_TIMEOUT;
429  }
430  /* buffer will be cleaned */
431  ctx->out_ifc_list[ifc].buffer_index = 0;
432  ctx->out_ifc_list[ifc].buffer_occupied = 0;
433  /* buffer was successfully sent but we still have current message pending/not stored
434  * it will be the first message in buffer */
435  if (ctx->out_ifc_list[ifc].bufferswitch == 1) {
436  insert_into_buffer(&ctx->out_ifc_list[ifc], data, size);
437  }
438  } else {
439  if (result == TRAP_E_TIMEOUT) {
440  ctx->counter_dropped_message[ifc]++;
441  }
442  if (trap_ctx_get_client_count(ctx, ifc) == 0) {
443  ctx->out_ifc_list[ifc].buffer_occupied = 0;
444  }
445  }
446  }
447 
448 fn_exit:
449  pthread_mutex_unlock(&ctx->out_ifc_list[ifc].ifc_mtx);
450  return result;
451 }

462 /**
463  * Function of reader-thread.
464  *
465  * \param[in] arg struct reader_threads_arg with context and thread id - used as the index in ctx->in_ifc_list, in_ifc_results
466  * \return NULL
467  */
468 void *reader_threads_fn(void *arg)
469 {
470  struct reader_threads_arg *argdata = (struct reader_threads_arg *) arg;
471  trap_ctx_priv_t *ctx = NULL;
472  int thread_id;
473  int retval;
474 #ifdef DISABLE_BUFFERING
475  uint32_t recvsize = 0;
476 #endif
477 
478  if (argdata == NULL) {
479  pthread_exit(NULL);
480  }
481  ctx = argdata->ctx;
482  thread_id = argdata->thread_index;
483  do {
484  sem_wait(&ctx->reader_threads[thread_id].sem);
485  if (ctx->terminated == 1) {
486  break;
487  }
488  /* call recv of my IFC and let it store results into multi-result array */
489 #ifndef DISABLE_BUFFERING
490  retval = trap_read_from_buffer(ctx, thread_id, (const void **) &ctx->in_ifc_results[thread_id].message,
491  &ctx->in_ifc_results[thread_id].message_size,
492  ctx->get_data_timeout);
493 #else
494  retval = ctx->in_ifc_list[thread_id].recv(ctx->in_ifc_list[thread_id].priv,
495  ctx->in_ifc_list[thread_id].buffer,
496  &recvsize,
497  ctx->get_data_timeout);
498  /* if sender uses buffering, we are loosing data! in addition, data can be corrupted in this case!!! */
499  ctx->in_ifc_results[thread_id].message_size = (uint16_t) recvsize;
500  ctx->in_ifc_results[thread_id].message = ctx->in_ifc_list[thread_id].buffer;
501 #endif
502 
503  ctx->in_ifc_results[thread_id].result_code = retval;
504  pthread_mutex_lock(&ctx->mut_sem_collector);
505  ctx->readers_count--;
506  if (ctx->readers_count == 0) {
507  /* the last reader wakes collector */
508  retval = sem_post(&ctx->sem_collector);
509  if (retval != 0) {
510  VERBOSE(CL_ERROR, "Waking up collector thread of multiread function failed. (%d)", retval);
511  }
512  }
513  pthread_mutex_unlock(&ctx->mut_sem_collector);
514  /* inform collector about finished job */
515 
516  if (ctx->terminated == 1) {
517  break;
518  }
519  } while (1);
520  free(arg);
521  pthread_exit(NULL);
522 }

531 static inline int trap_init_ifcs_timeouts(trap_ctx_priv_t *ctx)
532 {

535  idx = 0;
536  for (i = 0; i < ctx->num_ifc_out; i++) {
537  if ((ctx->out_ifc_list[i].timeout != TRAP_NO_AUTO_FLUSH) && (ctx->out_ifc_list[i].bufferswitch != 0)) {

541  out_ifc_timeout[idx].tm = ctx->out_ifc_list[i].timeout;
542  if (res == 0)
543  pthread_mutex_unlock(&ctx->out_ifc_list[i].ifc_mtx);
544  idx++;
545  }
546  }
547  // All changes updated, set to zero
548  if (pthread_rwlock_wrlock(&ctx->context_lock) != 0) {
549  VERBOSE(CL_ERROR, "Locking of context failed. %s", __func__);
550  }
551  ctx->ifc_change = 0;
552  pthread_rwlock_unlock(&ctx->context_lock);
553  return idx;
554 }

556 /**
557  * Handle the timeouts on output interfaces and flush buffer after timeout is reached.
558  *
559  *
560  * @return NULL
561  */
562 static void *trap_automatic_flush_thr(void *arg)
563 {
564  int i, n;
565  int64_t usec;
566  trap_ctx_priv_t *ctx = (trap_ctx_priv_t *) arg;
567 
568  n = trap_init_ifcs_timeouts(ctx);
569 
570  while (1) {
571  if (pthread_rwlock_rdlock(&ctx->context_lock) != 0) {
572  VERBOSE(CL_ERROR, "Locking of context failed. %s", __func__);
573  break;
574  }
575  if (ctx->terminated == 1) {
576  pthread_rwlock_unlock(&ctx->context_lock);
577  break;
578  }
579  // Checking if automatic flushing or buffering was changed or disabled
580  if (ctx->ifc_change == 1) {
581  pthread_rwlock_unlock(&ctx->context_lock);
582  n = trap_init_ifcs_timeouts(ctx);
583  } else {
584  pthread_rwlock_unlock(&ctx->context_lock);
585  }
586 
587  // Sort array by timeout if we have more than one output interface
588  if (n != 0) {
589  if (n > 1) {
590  qsort(ctx->ifc_autoflush_timeout, n, sizeof(struct out_ifc_timeout_s), compare_timeouts);
591  }
592  usec = ctx->ifc_autoflush_timeout[0].tm;
593  VERBOSE(CL_VERBOSE_LIBRARY, "Autoflush thread is going to sleep for %ld microseconds.", usec);
594  if (sleep(usec/1000000) != 0) {
595  if (pthread_rwlock_rdlock(&ctx->context_lock) != 0) {
596  VERBOSE(CL_ERROR, "Locking of context failed. %s", __func__);
597  break;
598  }
599  if (ctx->terminated == 1) {
600  pthread_rwlock_unlock(&ctx->context_lock);
601  break;
602  }
603  pthread_rwlock_unlock(&ctx->context_lock);
604  }
605  if (usleep(usec%1000000) == -1) {
606  if (pthread_rwlock_rdlock(&ctx->context_lock) != 0) {
607  VERBOSE(CL_ERROR, "Locking of context failed. %s", __func__);
608  break;
609  }
610  if (ctx->terminated == 1) {
611  pthread_rwlock_unlock(&ctx->context_lock);
612  break;
613  }
614  pthread_rwlock_unlock(&ctx->context_lock);
615  }
616 
617  // Check all interfaces if timeout has elapsed, otherwise break
618  for (i = 0; i < n; i++) {
619  ctx->ifc_autoflush_timeout[i].tm -= usec;
620  if (ctx->ifc_autoflush_timeout[i].tm == 0) {
621  pthread_mutex_lock(&ctx->out_ifc_list[i].ifc_mtx);
622  if (ctx->out_ifc_list[ctx->ifc_autoflush_timeout[i].idx].bufferflush == 0) {
623  pthread_mutex_unlock(&ctx->out_ifc_list[i].ifc_mtx);
624  // No event on the interface, flushing the buffer
625  trap_ctx_send_flush((trap_ctx_t *) ctx, i);
626  ctx->counter_autoflush[ctx->ifc_autoflush_timeout[i].idx]++;
627  }
628  else {
629  // Buffer was sent before timeout has elapsed, no need to flush the buffer
630  ctx->out_ifc_list[ctx->ifc_autoflush_timeout[i].idx].bufferflush = 0;
631  pthread_mutex_unlock(&ctx->out_ifc_list[i].ifc_mtx);
632  }
633  // Updating of timeout, it could be changing
634  pthread_mutex_lock(&ctx->out_ifc_list[i].ifc_mtx);
635  ctx->ifc_autoflush_timeout[i].tm = ctx->out_ifc_list[ctx->ifc_autoflush_timeout[i].idx].timeout;
636  pthread_mutex_unlock(&ctx->out_ifc_list[i].ifc_mtx);
637  } else {
638  continue;
639  }
640  }
641  } else {
642  // Sleep for defined time, default 2 seconds
643  if (sleep(TRAP_NO_IFC_SLEEP)) {
644  break;
645  }
646  }
647  }
648 
649  pthread_exit(NULL);
650 }

661 /** Initialization function.
662  * Create and initialize all interfaces. This function parses command-line
663  * arguments; it extracts arguments it needs to set up interfaces and returns
664  * the rest (argc and argv are modified).
665  * @param[in,out] argc Pointer to number of command-line arguments.
666  * @param[in,out] argv Command-line arguments.
667  * @param[out] ifc_spec Structure with specification of interface types and
668  * their parameters.
669  * @return Error code (0 on success)
670  */
671 int trap_parse_params(int *argc, char **argv, trap_ifc_spec_t *ifc_spec)
672 {
673  uint32_t i, j, ifc_count = 0;
674  char *ifc_spec_str = NULL;
675  char *ifc_type = NULL;
676  char *p;
677  int rv = TRAP_E_OK;
678 
679  if (ifc_spec == NULL) {
680  VERBOSE(CL_ERROR, "Bad pointer 'ifc_spec' passed to trap_parse_params().");
681  return TRAP_E_BAD_FPARAMS;
682  }
683 
684  /* initialization of ifc_spec */
685  memset(ifc_spec, 0, sizeof(trap_ifc_spec_t));
686 
687  // Look for -h (or --help) parameter
688  for (i = 0; i < *argc; i++) {
689  if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--help") == 0) {
690  if (i + 1 < *argc && ((strcmp(argv[i + 1], "trap") == 0) || (strcmp(argv[i + 1], "1") == 0))) {
691  trap_help_spec = 1;
692  }
693  // Remove parameter from argv and break
694  for (j = i; j + 1 < *argc; j++) {
695  argv[j] = argv[j + 1];
696  }
697  *argc -= 1;
698  return TRAP_E_HELP;
699  }
700  }
701 
702  // Extract -i parameter (interface specifier)
703  for (i = 0; i < *argc-1; i++) {
704  if (argv[i] && strcmp(argv[i], "-i") == 0) {
705  ifc_spec_str = argv[i+1];
706  // Remove parameters from argv
707  for (j = i; j + 2 < *argc; j++) {
708  argv[j] = argv[j + 2];
709  }
710  *argc -= 2;
711  break;
712  }
713  }
714  if (ifc_spec_str == NULL) {
715  //return trap_errorf(ctx, TRAP_E_BADPARAMS, "Interface specifier (option -i) not found.");
716  trap_last_error = TRAP_E_BADPARAMS;
717  trap_last_error_msg = "Interface specifier (option -i) not found.";
718  return TRAP_E_BADPARAMS;
719  }
720 
721  // Extract verbose level parameter (-v, -vv, -vvv)
722  for (i = 0; i < *argc; i++) {
723  // If param matches -v, -vv or -vvv, set verbosity level
724  if (strcmp(argv[i], "-v") == 0) {
725  trap_set_verbose_level(0);
726  } else if (strcmp(argv[i], "-vv") == 0) {
727  trap_set_verbose_level(1);
728  } else if (strcmp(argv[i], "-vvv") == 0) {
729  trap_set_verbose_level(2);
730  } else {
731  continue;
732  }
733  // Remove parameter from argv and break
734  for (j = i; j + 1 < *argc; j++) {
735  argv[j] = argv[j + 1];
736  }
737  *argc -= 1;
738  break;
739  }
740 
741  /* count the number of IFC parameters, the format is:
742  * type1:param1:param1_2,type2:,type3:param3 */
743  p = ifc_spec_str;
744  while (p != NULL) {
745  p = strchr(p, TRAP_IFC_DELIMITER);
746  if (p != NULL && *p == TRAP_IFC_DELIMITER) {
747  p++;
748  }
749  ifc_count++;
750  }
751 
752  ifc_spec->types = calloc(ifc_count + 1, sizeof(*ifc_spec->types));
753  if (ifc_spec->types == NULL) {
754  return TRAP_E_MEMORY;
755  }
756  ifc_spec->params = calloc(ifc_count, sizeof(char *));
757  if (ifc_spec->params == NULL) {
758  free(ifc_spec->types);
759  ifc_spec->types = NULL;
760  return TRAP_E_MEMORY;
761  }
762 
763  p = ifc_spec_str;
764  for (i = 0; i < ifc_count; i++) {
765  /* set IFC type and skip to params */
766  if (p != NULL) {
767  ifc_spec->types[i] = p[0];
768  if (strlen(p) >= 2 && p[1] == TRAP_IFC_PARAM_DELIMITER) {
769  /* there is a param string to parse */
770  p += 2;
771  p = get_param_by_delimiter(p, &ifc_spec->params[i], TRAP_IFC_DELIMITER);
772  } else {
773  /* param was not passed, set an empty one */
774  ifc_spec->params[i] = strdup("");
775  }
776  } else {
777  /* unexpected error, p shouldn't be NULL */
778  VERBOSE(CL_ERROR, "Bad IFC_SPEC '%s'. See -h trap for help.", ifc_spec_str);
779  ifc_spec->params[i] = strdup("");
780  }
781  if (ifc_spec->params[i] == NULL) {
782  VERBOSE(CL_ERROR, "Allocation failed.");
783  rv = TRAP_E_MEMORY;
784  goto clean_on_fail;
785  }
786  }
787 
788  /* check for unsupported IFCs */
789  for (i = 0; ifc_spec->types[i] != 0; ++i) {
790  for (ifc_type = trap_ifc_type_supported; *ifc_type != 0; ++ifc_type) {
791  if (*ifc_type == ifc_spec->types[i]) {
792  break;
793  }
794  }
795  if (*ifc_type == 0) {
796  /* not found */
797  VERBOSE(CL_ERROR, "Unsupported IFC type '%c'.", ifc_spec->types[i]);
798  rv = TRAP_E_BADPARAMS;
799  goto clean_on_fail;
800  }
801  }
802 
803  trap_last_error = TRAP_E_OK;
804  trap_last_error_msg = default_err_msg[TRAP_E_OK];
805  return TRAP_E_OK;
806 
807 clean_on_fail:
808  for (i = 0; i < ifc_count; i++) {
809  /* set IFC type and skip to params */
810  if (ifc_spec->params[i] != NULL) {
811  free(ifc_spec->params[i]);
812  ifc_spec->params[i] = NULL;
813  }
814  }
815  if (ifc_spec->types != NULL) {
816  free(ifc_spec->types);
817  }
818  if (ifc_spec->params) {
819  free(ifc_spec->params);
820  }
821  memset(ifc_spec, 0, sizeof(trap_ifc_spec_t));
822  return rv;
823 }

1091 /**
1092  * Print helper for formatting text of help.
1093  *
1094  * \param[in] s String to print (e.g. description).
1095  * \param[in] align The number of spaces that should be add after line-break.
1096  * \param[in] cut Approximate (minimal) length of string that is printed on the line.
1097  * Line-break is added if needed, after the current word.
1098  */
1099 static void print_aligned(const char *s, uint16_t align, uint16_t cut)
1100 {
1101  int32_t tmp = 0, rest, written;
1102  const char *p = s;
1103 
1104  rest = strlen(s);
1105  if (rest < cut) {
1106  printf("%s\n", s);
1107  } else {
1108  written = 0;
1109  while (rest > 0) {
1110  /* print alignment spaces */
1111  if (p != s) {
1112  printf("%*s", align, " ");
1113  }
1114  /* skip leading spaces of text */
1115  while ((*p == ' ' || *p == '\t') && *p != 0) {
1116  p++;
1117  rest--;
1118  }
1119  /* find word boundary */
1120  if (rest > cut) {
1121  for (tmp = cut; tmp > 0; tmp--) {
1122  switch (p[tmp]) {
1123  case '.':
1124  case ',':
1125  case ';':
1126  /* let this character on the end of line */
1127  tmp += 1;
1128  goto write;
1129  case ' ':
1130  case '\t':
1131  case '\n':
