72 /**
73  * Receive one message via this IFC.
74  *
75  * This function is called from trap_read_from_buffer() when there is a
76  * need to get new data.
77  *
78  * \param[in] p pointer to IFC's private memory allocated by constructor
79  * \param[out] d pointer to memory where this IFC can write received message
80  * \param[out] s size (in bytes) of received message (must be set by this IFC)
81  * \param[in] t timeout, see \ref trap_timeout
82  * \returns TRAP_E_OK on success
83  */
84 typedef int (*ifc_recv_func_t)(void *p, void *d, uint32_t *s, int t);
85 
86 /**
87  * Send one message via this IFC.
88  *
89  * This function is called from trap_store_into_buffer() when there is a
90  * need to send new data.
91  *
92  * \param[in] p pointer to IFC's private memory allocated by constructor
93  * \param[in] d pointer to message that will be sent
94  * \param[in] s size (in bytes) of message that will be sent
95  * \param[in] t timeout, see \ref trap_timeout
96  * \returns TRAP_E_OK on success
97  */
98 typedef int (*ifc_send_func_t)(void *p, const void *d, uint32_t s, int t);

139 /**

142  * \param[in] priv pointer to IFC's private memory allocated by constructor
143  * \returns pointer to char (private memory of the ifc - do not free!).
