59 /**
60  * \brief Return syslog and output level based on given verbose level
61  *
62  * \param level Verbose level
63  * \param str_level String that is set according to given verbose level
64  */
65 static void get_level(int level, char **str_level)
66 {
67  static char *error="ERROR";
68  static char *warning="WARNING";
69  static char *notice="NOTICE";
70  static char *verbose="VERBOSE";
71  static char *adv_verbose="ADVANCED VERBOSE";
72  static char *library="LIBRARY VERBOSE";
73 
74  switch (level) {
75  case CL_ERROR:
76  *str_level = error;
77  break;
78  case CL_WARNING:
79  *str_level= warning;
80  break;
81  case CL_VERBOSE_OFF:
82  *str_level= notice;
83  break;
84  case CL_VERBOSE_BASIC:
85  *str_level= verbose;
86  break;
87  case CL_VERBOSE_ADVANCED:
88  *str_level= adv_verbose;
89  break;
90  case CL_VERBOSE_LIBRARY:
91  *str_level= library;
92  break;
93  default:
94  *str_level= notice;
95  }
96 }

98 /**
99  * \brief send verbose message to stderr
100  *
101  * send verbose message to stderr. may change in future. don't use it directly
102  *
103  * \param level importance level
104  * \param string format string, like printf function
105  */
106 void trap_verbose_msg(int level, char *string)
107 {
108  char *strl;
109  get_level(level, &strl);
